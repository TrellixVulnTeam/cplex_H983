19: int d1 = ...;
20: int d2 = ...;
21: 
22: range R1 = 1..d1;
23: range R2 = 1..d2;
24: 
25: range dim  = 1..d1*d2;
26: int Costs[i in R2][j in R1] = ...;
27: 
28: dvar float X[R2][R1];
29: 
30: dvar boolean Y[R1];
31: 
32: int bendersPartition[i in R2][j in R1] = i;
33: int bendersPartition2[i in R2] = i;
34: 
35: minimize sum(i in R2, j in R1) Costs[i][j]*X[i][j] + sum(i in R1) Y[i];
36: subject to{
37: forall(i in R2)
38:   sum(j in R1) X[i][j] ==1;
39: forall(i in R2, j in R1)
40:   X[i][j] - Y[j] <= 0;
41: }
